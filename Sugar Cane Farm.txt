function turtle.blockDown()
  local success, block = turtle.inspectDown()
  if block.name == nil then
    return nil
  else
    return block.name
  end    
end

  for x = 1, 7 do
  turtle.forward()

  turtle.dig()

  end
  end
  turtle.turnLeft()

  turtle.dig()

  turtle.suckDown
